1 #!/bin/env bash
 2 #SBATCH -J local_scratch_example
 3 #SBATCH -o local_scratch_example.o_%j
 4 #SBATCH -e local_scratch_example.e_%j
 5 #SBATCH --mem 10G
 6 #SBATCH --cpus-per-task 1
 7 #SBATCH --ntasks 1
 8 #SBATCH --nodes 1
 9 #SBATCH --time 00:10:00
10 
11 # Load module commands go here
12 # module load XXXX/XXXXXXX 
13 # module load YYYY/YYYYYY 
14 
15 # list all loaded modules
16 module list
17 
18 # Here we are making dummy "input" files testing1two3four5six_1 through testing1two3four5six_9,
19 # but you wouldn't have to do this for your real script
20 seq 1 9 | xargs -I {} touch testing1two3four5six_{}
21 
22 MY_STARTING_DIR=$PWD
23 
24 MY_LOG_FILE=$MY_STARTING_DIR/local_dir_job_${SLURM_JOB_ID}.log
25 
26 # Make unique directory
27 LOCAL_DIR=/local/scratch/$USER/$SLURM_JOB_ID/testingDIR
28 
29 # Create the starting directory 
30 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
31 echo "creating $LOCAL_DIR on $HOSTNAME"                                >> $MY_LOG_FILE
32 mkdir -p $LOCAL_DIR
33 
34 # Log how much space is on /local/scratch (then you can blame someone else if it is already full)
35 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
36 echo "Space on $HOSTNAME's /local/scratch before this job:"            >> $MY_LOG_FILE
37 df -h $LOCAL_DIR                                                       >> $MY_LOG_FILE
38 
39 # You probably don't need to copy files to /local/scratch, but assuming you do ....
40 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
41 echo "Copying input files to $LOCAL_DIR on $HOSTNAME"                  >> $MY_LOG_FILE
42 
43 cp testing123456* $LOCAL_DIR
44 
45 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
46 echo "Space on $HOSTNAME's /local/scratch after copying files:"        >> $MY_LOG_FILE
47 df -h $LOCAL_DIR                                                       >> $MY_LOG_FILE
48 
49 # Change to local directory
50 cd $LOCAL_DIR
51 
52 echo "---- START OF ANALYSIS PIPELINE -------------------------------" >> $MY_LOG_FILE
53 # Here is where your pipeline code would go
54 
55 # You may have many result files, but we're just creating one for the demo
56 RESULT_FILE=local_scratch_info_${SLURM_JOB_ID}.txt
57 
58 # replace the following with your analysis steps
59 hostname  > $RESULT_FILE
60 pwd      >> $RESULT_FILE
61 ls -l    >> $RESULT_FILE
62 df -h .  >> $RESULT_FILE
63 
64 echo "---- END OF ANALYSIS PIPELINE -------------------------------" >> $MY_LOG_FILE
65 
66 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
67 echo "Space on $HOSTNAME's /local/scratch after running the pipeline:" >> $MY_LOG_FILE
68 df -h /local/scratch                                                   >> $MY_LOG_FILE
69 
70 # Copy results to a "permanent" non-local directory 
71 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
72 echo "Directory listing of $LOCAL_DIR: "                               >> $MY_LOG_FILE
73 ls                                                                     >> $MY_LOG_FILE
74 echo "Copying $RESULT_FILE to $MY_STARTING_DIR "                       >> $MY_LOG_FILE
75 
76 cp $RESULT_FILE $MY_STARTING_DIR
77 
78 # Go back to starting directory
79 cd $MY_STARTING_DIR
80 
81 # Clean up (Remove all of the files we just created)
82 echo "---------------------------------------------------------------"     >> $MY_LOG_FILE
83 echo "Removing $LOCAL_DIR, to leave space on /local/scratch for next time" >> $MY_LOG_FILE
84 rm -rf $LOCAL_DIR
85 
86 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
87 echo "Space on $HOSTNAME's /local/scratch after deleting $LOCAL_DIR:"  >> $MY_LOG_FILE
88 df -h /local/scratch                                                   >> $MY_LOG_FILE
89 
90 echo "---------------------------------------------------------------" >> $MY_LOG_FILE
91 echo "Finished deleting $LOCAL_DIR" >> $RESULT_FILE
92 
93 echo "** FINISHED ***************************************************" >> $MY_LOG_FILE
94 
95 # Removing our dummy input files, but you probably wouldn't need this for a real script
96 rm testing1two3four5six_*
